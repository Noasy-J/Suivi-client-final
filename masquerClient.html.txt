  <script>
    const $ = id => document.getElementById(id);

    function showError(msg){
      const box = $('errorBox');
      box.textContent = msg;
      box.style.display = 'block';
    }

    function renderClientsDropdown(res) {
      const sel = $('clientSelect');
      sel.innerHTML = '';
      const list = (res && res.clients) || [];
      if (!list.length) {
        const o = document.createElement('option');
        o.value = '';
        o.textContent = '‚Äî Aucun client (Client R√©gulier) ‚Äî';
        sel.appendChild(o);
        return;
      }
      list.forEach(c => {
        const o = document.createElement('option');
        o.value = c.key; o.textContent = c.label;
        sel.appendChild(o);
      });
    }

    function renderMaskedList(res) {
      const cont = $('maskedList');
      cont.innerHTML = '';
      const arr = (res && res.masked) || [];
      if (!arr.length) {
        cont.innerHTML = '<div class="empty">Aucun client masqu√©.</div>';
        return;
      }
      arr.forEach(m => {
        const div = document.createElement('div');
        div.className = 'item';
        const left = document.createElement('div');
        left.innerHTML = '<b>'+m.label+'</b>' + (m.untilISO ? ' <small>(jusqu‚Äôau '+m.untilISO+')</small>' : '');
        const btn = document.createElement('button');
        btn.className = 'btn-danger';
        btn.textContent = 'D√©masquer';
        btn.onclick = () => google.script.run.withSuccessHandler(refreshUI).uiUnmaskClient_(m.key);
        div.appendChild(left); div.appendChild(btn);
        cont.appendChild(div);
      });
    }

    function refreshUI() {
      // min du calendrier = aujourd‚Äôhui
      $('dateInput').setAttribute('min', new Date().toISOString().slice(0,10));

      google.script.run
  .withSuccessHandler((res) => {
    if (!res || res.ok === false) {
      // Affiche l‚Äôerreur renvoy√©e c√¥t√© serveur
      const box = document.getElementById('errorBox');
      if (box) { box.textContent = res && res.message ? res.message : 'Erreur serveur.'; box.style.display = 'block'; }
      return;
    }
    // ok => on masque l‚Äôerreur et on remplit le dropdown
    const box = document.getElementById('errorBox');
    if (box) box.style.display = 'none';
    renderClientsDropdown(res);
    renderMaskedList(res);
    document.getElementById('dateInput').setAttribute('min', new Date().toISOString().slice(0,10));
  })
  .withFailureHandler((err) => {
    const box = document.getElementById('errorBox');
    if (box) { box.textContent = (err && err.message) ? err.message : 'Erreur Apps Script.'; box.style.display = 'block'; }
  })
  .uiListClientsPourMasquage_();


    function maskByDate() {
      const key = $('clientSelect').value;
      const d   = $('dateInput').value;
      if (!key) return alert('S√©lectionnez un client.');
      if (!d)   return alert('Choisissez une date.');
      google.script.run.withSuccessHandler((r)=>{
        if (!r || r.ok === false) return alert((r && r.message) || 'Erreur.');
        refreshUI();
      }).uiMaskClient_({key, mode:'date', dateISO:d});
    }

    function maskUntilNext() {
      const key = $('clientSelect').value;
      if (!key) return alert('S√©lectionnez un client.');
      google.script.run.withSuccessHandler((r)=>{
        if (!r || r.ok === false) return alert((r && r.message) || 'Erreur.');
        refreshUI();
      }).uiMaskClient_({key, mode:'next'});
    }

    document.addEventListener('DOMContentLoaded', refreshUI);
  </script>
